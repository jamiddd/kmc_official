<script>
    $(document).ready(function(){
        $('.tooltipped').tooltip();
    });

    $(".res").click(function () {
        for (var i = 0; i < 6; i++){
            $("#apply-form")[0].reset();
        }
    });

    $(document).ready(function(){
        $(".sticky-news").hover(function(){
            $(this).removeClass('weird');
        }, function(){
            // $(this).addClass('weird');
        });
    });

    var height =  window.innerHeight || document.documentElement.clientHeight || document.body.clientHeight;

    $(window).scroll(function(){
        var newsSection = $('.sticky-news'),
        scroll = $(window).scrollTop();

        if (scroll >= height/2) 
            newsSection.addClass('weird');
        else 
            newsSection.removeClass('weird');
    });

    
    $(window).scroll(function(){
        var applyButton = $('.fixed-apply-now'),
        scroll = $(window).scrollTop();

        if (scroll <= height/2 || ($(window).scrollTop() + $(window).height() > ($(document).height() - height/2)))
            applyButton.addClass('weird-again');
        else 
            applyButton.removeClass('weird-again');
    });

</script>
